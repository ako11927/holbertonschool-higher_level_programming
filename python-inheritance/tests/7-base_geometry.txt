#!/usr/bin/python3
BaseGeometry = __import__('7-base_geometry').BaseGeometry

bg = BaseGeometry()

# Test cases from your screenshot
test_cases = [
    ("name", "John", TypeError, "name must be an integer"),
    ("age", 0, ValueError, "age must be greater than 0"),
    ("distance", -4, ValueError, "distance must be greater than 0"),
    ("age", 1, None, "Should not raise exception"),
    ("age", -4, ValueError, "age must be greater than 0"),
    ("age", "4", TypeError, "age must be an integer"),
    ("age", (4, 1), TypeError, "age must be an integer"),
    ("age", [3], TypeError, "age must be an integer"),
    ("age", True, TypeError, "age must be an integer"),
    ("age", [3, 4], TypeError, "age must be an integer"),
    ("age", None, TypeError, "age must be an integer"),
]

for name, value, expected_exception, description in test_cases:
    try:
        bg.integer_validator(name, value)
        if expected_exception is None:
            print(f"✓ {description}")
        else:
            print(f"✗ {description}: Expected {expected_exception.__name__} but got no exception")
    except Exception as e:
        if type(e) == expected_exception:
            print(f"✓ {description}: Got expected error: {e}")
        else:
            print(f"✗ {description}: Expected {expected_exception.__name__} but got {type(e).__name__}: {e}")
